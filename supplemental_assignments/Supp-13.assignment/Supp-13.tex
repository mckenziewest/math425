\textbf{(Supp-13)} Let $G$ be a group and $g,h\in G$.
\begin{enumerate}[label=(\alph*)]
	\item Assume $gh=hg$ and that $o(g)$ and $o(h)$ are finite, show that $o(gh)$ is also finite.
	\vfill
	\item Show that (a) fails if $gh\neq hg$ by considering the group of $2\times2$ invertible matrices under the operation of matrix multiplication, $GL_2(\mathbb{R})$, and the elements $$g=\begin{bmatrix}0&-1\\1&0\end{bmatrix}\quad\text{ and }   h=\begin{bmatrix}0&1\\-1&-1\end{bmatrix}.$$
	\vskip 2in
\end{enumerate}
