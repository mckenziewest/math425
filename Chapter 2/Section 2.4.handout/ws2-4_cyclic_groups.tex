\section*{Subgroups Generated by Elements}
\slide{
	\begin{thm}{2.4.1}
		Let $g$ be an element of a group $G$, and write
			\[\langle g\rangle =\{g^k:k\in\Z\}.\]
		Then $\langle g\rangle$ is a subgroup of $G$, and $\langle g\rangle\subseteq H$ for every subgroup $H$ of $G$ with $g\in H$.
	\end{thm}
	\vfill

\section*{Orders of Groups and Elements}
\slide{
	\begin{defn}
		If $G$ is a finite group, the \emph{order of a group} $G$ is denoted $|G|$ and is the cardinality of the set $G$.
		
		The \emph{order of an element} $g\in G$ is denoted $|g|$ or $o(g)$ and equals the smallest positive integer $n$ such that $g^n=e$.
	\end{defn}

\slide{
	\begin{thm}{2.4.2}
		Let $g\in G$ with $o(g)=n$.  Then
		\enumarabic{
			\setlength{\itemsep}{2em}
			\item $g^k=e$ if and only if $n|k$.
			\item $g^k=g^m$ if and only if $k\equiv m\pmod n$
			\item $\langle g\rangle=\{e,g,g^2,\dots, g^{n-1}\}$ where $e,g,g^2,\dots,g^{n-1}$ are all distinct.
		}
	\end{thm}
	* The proof is in the textbook, and uses laws of exponents.
}

	\begin{thm}{2.4.3}
		Let $G$ be a group and let $g\in G$ satisfy $o(g)=\infty$.  Then 
			\enumarabic{
				\setlength{\itemsep}{2em}
				\item $g^k=e$ if and only if $k=0$.
				\item $g^k=g^m$ if and only if $k=m$.
				\item $\langle g\rangle =\{\dots,g^{-2},g^{-1},e,g,g^2,\dots\}$ where the $g^i$ are distinct.

	\end{thm}

		\begin{statementblock}{Theorem}
			If $\gamma=(k_1\ k_2\ \dots\ k_r)$ is an $r$-cycle in $S_n$, then $o(\gamma)=r$.
		\end{statementblock}
		\begin{question} You proved $(1\ 2\ 3\ \dots\ n)$ has order $n$ for homework, how might you extend this proof to work for any cycle?
		\end{question}

		\begin{thm}{2.4.4}
			If $\gamma=\sigma_1\sigma_2\dots\sigma_r$ where $\sigma_i$ are disjoint cycles, then
				\[o(\gamma)=\operatorname{lcm}(o(\sigma_1),o(\sigma_2),\dots,o(\sigma_r)).\]
		\end{thm}

	\section*{Finite Cyclic Groups and Their Subgroups}
\slide{
	\begin{statementblock}{The Fundamental Theorem of Finite Cyclic Groups (Theorem 2.4.9)}
		Let $G=\langle g\rangle$ be a cyclic group of order $n$.
		\enumarabic{
			\item If $H$ is a subgroup of $G$, then $H=\langle g^d\rangle$ for some $d|n$. Hence $|H|$ divides $n$.
			\item Conversely if $k|n$, then $\langle g^{n/k}\rangle$ is the unique subgroup of $G$ of order $k$.
		}
	\end{statementblock}
	\vskip 1in
}

	\section*{Groups Generated by Several Elements}
		\begin{exa}
			The Klein-4 Group $K_4$:
			$$\Z_2\times \Z_2=\{(0,0),(1,0),(0,1),(1,1)\}=\langle(\overline{0},\overline{1}),(\overline{1},\overline{0})\rangle$$
		\end{exa}
		\vfill
		\begin{exa}
			The symmetric group $S_3$:
			Let $\sigma = (1\ 2\ 3)$ and $\tau = (1\ 2)$.  Then every element of $S_3$ can be written as $\sigma^i\tau^j$ for some $i,j$.
			
			\begin{multicols}{3}
				
			\begin{itemize}
				\item $\varepsilon=\sigma^0\tau^0$
				\item $(1\ 2)=\sigma^0\tau^1$
				\item $(1\ 3)=\sigma^1\tau^1$
				\item $(2\ 3)=\sigma^2\tau^1$
				\item $(1\ 2\ 3)=\sigma^1\tau^0$
				\item $(1\ 3\ 2)=\sigma^2\tau^0$
			\end{itemize}
			\end{multicols}
			
			Thus $S_3=\langle \sigma,\tau\rangle$.
		\end{exa}
	}
\slide{
	\begin{defn}
		In general, if $X$ is a nonempty subset of a group $G$, then the \emph{subgroup of $G$ generated by $X$} is defined as	
		\[\begin{array}{rcl}
		\langle X\rangle &=& \{\text{products of powers (not nec. distinct) of elements of X}\}\\
		&=&\{x_1^{k_1}x_2^{k_2}\cdots x_m^{k_m}\ |\ x_i\in X,\ k_i\in \Z,\ m\geq 1\}
		\end{array}\]
	We will always have $\langle X\rangle\leq G$.
	\end{defn}
